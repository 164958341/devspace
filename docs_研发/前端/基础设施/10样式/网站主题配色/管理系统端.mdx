export const AppLayout = (props) => {
  let HeaderComponent;
  if (props.header) {
    HeaderComponent = <div style={{ height: "80px" }}>{props.header}</div>;
  }
  let FooterComponent;
  if (props.footer) {
    FooterComponent = <div style={{ height: "80px" }}>{props.footer}</div>;
  }
  let MainComponent;
  if (props.content) {
    MainComponent = <div style={{ flex: "1 auto" }}>{props.content}</div>;
  }
  return (
    <div
      style={{
        margin: "20px 0",
        width: "1280px",
        height: "960px",
        display: "flex",
        flexDirection: "column",
      }}
    >
      <HeaderComponent></HeaderComponent>
      <MainComponent></MainComponent>
      <FooterComponent></FooterComponent>
    </div>
  );
};
